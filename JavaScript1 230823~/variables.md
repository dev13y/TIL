# variables

## `JavaScript<variables>`
```html
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>변수</title>
    <script>
        // 명시적 선언
        let num=10, pi=3.14;
        let ch = 'a';
        let name = "홍길동";

        // 명시적 선언없이 바로 값 저장
        address = "서울시 1번지";
        nation = '대한민국'; // 문자열에 작은 따옴표 / 큰 다옴표 모두 사용 가능

        // 변수값 출력
        console.log(num);
        console.log(pi);
        console.log(ch);
        console.log(name);
        console.log(address);
        console.log(nation);

        // 문자열 + 변수 + 태그 연결해서 출력
        document.write("이름 : " + name + "<br>");
        document.write("주소 : " + address);
    </script>
</head>
<body>
    
</body>
</html>
```
